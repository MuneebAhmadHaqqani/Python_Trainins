create our deep learning model¶

In [69]:
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense , Conv2D , MaxPool2D,Flatten
In [70]:
model=Sequential()

#CNN layers
model.add(Conv2D(filters=32,kernel_size=(4,4),input_shape=(32, 32, 3),activation="relu")) # 32 types of filters are used here , with its size 4*4
model.add(MaxPool2D())  # This will agin do filtering with max values, there Avg pool also 
model.add(Flatten()) # this will convert the values to one row and many columns before feeding to ANN

# ANN layers
model.add(Dense(120,activation="relu")) # HL
model.add(Dense(10,activation="softmax")) # O/p layer  function is softmax because here multi class in the output
In [71]:
model.compile(optimizer="adam",loss="categorical_crossentropy",metrics=["accuracy"]) # more than 2 class hence categorical ....
In [72]:
model.fit(scaled_xtrain,ytrain,epochs=10,batch_size=1) # batch size need to be added ----> batch_size=1
